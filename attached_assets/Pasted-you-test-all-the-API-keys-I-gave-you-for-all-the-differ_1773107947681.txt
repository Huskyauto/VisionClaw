import os

    key = os.getenv("OPENAI_API_KEY")

        record("openai", False, "NO KEY SET")

    try:
        _ = client.models.list()
        record("openai", True, "models.list() succeeded")

        record("openai", False, str(e))

# file: verify_llm_keys.py
import os
import asyncio
import httpx

PROVIDERS = {}

def provider(name):
    def wrapper(fn):
        PROVIDERS[name] = fn
        return fn
    return wrapper

RESULTS = {}

def record(name, ok, detail):
    RESULTS[name] = {"ok": ok, "detail": detail}

@provider("openai")
async def check_openai():
    key = os.getenv("OPENAI_API_KEY")
    if not key:
        record("openai", False, "NO KEY SET")
        return
    from openai import OpenAI
    client = OpenAI(api_key=key)
    try:
        client.models.list()
        record("openai", True, "models.list ok")
    except Exception as e:
        record("openai", False, repr(e))

@provider("anthropic")
async def check_anthropic():
    key = os.getenv("ANTHROPIC_API_KEY")
    if not key:
        record("anthropic", False, "NO KEY SET")
        return
    import anthropic
    client = anthropic.Anthropic(api_key=key)
    try:
        client.models.list()
        record("anthropic", True, "models.list ok")
    except Exception as e:
        record("anthropic", False, repr(e))

@provider("perplexity")
async def check_perplexity():
    key = os.getenv("PERPLEXITY_API_KEY")
    if not key:
        record("perplexity", False, "NO KEY SET")
        return
    headers = {"Authorization": f"Bearer {key}", "Content-Type": "application/json"}
    data = {
        "model": "sonar-small-online",

        else:
            record("perplexity", False, f"HTTP {r.status_code}: {r.text[:200]}")
    except Exception as e:
        record("perplexity", False, repr(e))

# add more providers in the same pattern...

async def main():
    tasks = [fn() for fn in PROVIDERS.values()]
    await asyncio.gather(*tasks)

    print("\n=== LLM KEY CHECK ===")
    for name, res in RESULTS.items():
        status = "OK" if res["ok"] else "FAIL"
        print(f"{name:12} {status:4} - {res['detail']}")

if __name__ == "__main__":
    asyncio.run(main())
